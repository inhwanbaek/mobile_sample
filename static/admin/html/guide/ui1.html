							ui.alert("공통 알럿창입니다.",{ // 알럿 옵션들
								tit:"알럿타이틀",
								ycb:function(){
									console.log("알럿확인결과");
								},
								ybt:"확인" // 기본값 "확인"
							});

							ui.confirm("공통 컨펌창 입니다.",{ // 컨펌 창 옵션들
								tit:"타이틀",
								ycb:function(){
									console.log("컨펌확인결과");
								},
								ncb:function(){
									console.log("컨펌취소결과");
								},
								ybt:"확인", // 기본값 "확인"
								nbt:"취소"  // 기본값 "취소"
							});
